	jogo_id INTEGER NOT NULL,
	PRIMARY KEY (jogador_id),
	PRIMARY KEY (jogador_id, jogo_id),
	FOREIGN KEY (jogado_id) REFERENCES Jogador(jogador_id)

CREATE TABLE Jogo (
	jogo_id INTEGER IDENTITY(1,1) primary key,
	informacao VARCHAR(50) NOT NULL --Perdido, ganho ou abandonado
	estado VARCHAR(64),--cada valor da variavel é uma casa do tabuleiro de um determinado jogo
	PRIMARY KEY (jogo_id)
)


CREATE TABLE Imagem (
	imagem_id INTEGER IDENTITY(1,1) NOT NULL,

	local_imagem VARCHAR(200) NOT NULL,

	FOREIGN KEY (jogado_id) REFERENCES Jogador(jogador_id)
